Summavektori $\va + \vb$ l‰htee vektorin $\va$ kanssa samasta pisteest‰ ja p‰‰ttyy vektorin $\vb$ k‰rkeen. Summavektori $\va + \vb$ saadaan suoraan laskemalla vektorit $\va$ ja $\vb$ yhteen komponenteittain:

	\begin{align*}
	\va + \vb
	&= (8\vi +2\vj)+(-5\vi+3\vj) \\
	&= 8\vi +2\vj -5\vi +3\vj \\
	&= 8\vi -5\vi +2 \vj+3\vj \\
	&= 3\vi + 5\vj. 
	\end{align*}
	
\maaritelma[Summavektori]{Vektorien $\vv=x_1\vi+y_1\vj$ ja $\vw=x_2\vi+y_2\vj$ \textbf{summavektori} on vektori $\vv+\vw=(x_1+x_2)\vi+(y_1+y_2)\vj$.}



	\begin{teht} Tutkitaan vektoreita $\vv=-2i+j$ ja $\vw=4i+4j$.
		\begin{aakkosta*}
		\item Piirr‰ summavektori $\vv+\vw$ koordinaatistoon vektorien $\vv$ ja $\vw$ avulla. 
		\item Ilmoita vektori $\vv+\vw$ vektorien $\vi$ ja $\vj$ avulla.
		\item Tarkista tuloksesi laskemalla yhteen vektorit $\vv$ ja $\vw$ komponenteittain.
		\end{aakkosta*}
	\end{teht}





	\begin{teht} Tutkitaan vektoreita $\vu=i+2\vj$, $\vv=-3i+j$ ja $\vw=3i+-5j$.
		\begin{aakkosta*}
		\item Piirr‰ vektori $\vu+\vv+\vw$ koordinaatistoon vektorien $\vu, \vv$ ja $\vw$ avulla. 
		\item Piirr‰ vektori $\vv+\vu+\vw$ koordinaatistoon vektorien $\vu, \vv$ ja $\vw$ avulla. 
		\item Piirr‰ vektori $\vw+\vu+\vv$ koordinaatistoon vektorien $\vu, \vv$ ja $\vw$ avulla. 
		\item Mit‰ huomaat?
		\end{aakkosta*}
	\end{teht}

\maaritelma[Vektorin kertominen reaaliluvulla]{Kun vektoria $\vv=x\vi+y\vj$ kerrotaan reaaliluvulla $r$, saadaan vektori $r\vv=r(x\vi+y\vj)=(rx)\vi+(ry)\vj$.}


Kun siis esimerkiksi vektoria $\vv=-4\vi+3\vj$ kerrotaan luvulla $2$, saadaan vektori 

\begin{align*}
2\vv
=2 \cdot (-4\vi+3\vj)
= 2\cdot (-4) \vi + 2\cdot 3 \vj
= -8\vi + 6 \vj.
\end{align*}

\subsection{Erotus}


	
	\begin{teht} Tutkitaan vektoreita $\vv=-2\vi+3\vj$ ja $\vw=2\vi-3\vj$.

		\item M‰‰rit‰ summavektori $\vv+\vw$ laskemalla yhteen vektorit $\vv$ ja $\vw$ komponenteittain. 
		\item Piirr‰ vektori $\vv+\vw$ koordinaatistoon vektorien $\vv$ ja $\vw$ avulla. 
		\end{aakkosta*}
	\end{teht}

Edellisen teht‰v‰n vektorit $\vv$ ja $\vw$ ovat toistensa vastavektoreita. N‰in ollen voidaan merkit‰ $\vw=-\vv$, ja summavektori $\vv+\vw$ saadaan muotoon $\vv+\vw=\vv +(-\vv)=\vv-\vv=0$. 



	\begin{teht} Tutkitaan vektoreita $\vv=$ ja $\vw=$
		\begin{aakkosta*}
		\item Piirr‰ vektori $\vv$ koordinaatistoon.
		\item Muodosta vektorin $\vw$ vastavektori $-\vw$.
		\item Piirr‰ vektori $-\vw$ koordinaatistoon siten, ett‰ ett‰ se alkaa vektorin $\vv$ k‰rjest‰.

		\item Ilmoita vektori $\vv+(-\vw)$ vektorien $\vi$ ja $\vj$ avulla.
		\end{aakkosta*}
	\end{teht}
