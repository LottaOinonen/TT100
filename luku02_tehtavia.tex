\begin{teht} Tarkastele vektoreita $\va, \vb \neq \bar{0}$. Tiedet‰‰n, ett‰ 
$$
\vb-3\va=5(\va-\vb).
$$
\begin{aakkosta*}
\item Osoita, ett‰ vektorit $\va$ ja $\vb$ ovat yhdensuuntaiset.
\item Ovatko vektorit $\va$ ja $\vb$ saman- vai vastakkaissuuntaiset?
\end{aakkosta*}
\end{teht}


\begin{teht} Tarkastele vektoreita $\va, \vb \neq \bar{0}$. Oletetaan, ett‰ vektorit eiv‰t ole yhdensuuntaiset. Osoita, ett‰ vektori $3\va+\vb$ ei ole yhdensuuntainen vektorin $\va$ kanssa.
\end{teht}

\begin{teht} Tarkastele vektoreita $\va, \vb \neq \bar{0}$, $\va \nparallel \vb$. Osoita, ett‰ vektori $-2\va+15\vb$ ei ole nollavektori.
\end{teht}

\begin{teht} Mit‰ tied‰t vektorien $\va$ ja $\vb$ suunnasta ja pituudesta, kun 
\begin{align*}
2(2\va-3\vb)-5\va-(3\va-4\vb)=\bar{0}.
\end{align*}
\end{teht}
